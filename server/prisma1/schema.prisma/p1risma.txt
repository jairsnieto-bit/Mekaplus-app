generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model User {
  id        String   @id @default(uuid())
  name      String
  email     String   @unique
  password  String
  role      Role     @default(OPERATOR)
  active    Boolean  @default(true)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
  guides    Guide[]
  
  @@index([email])
  @@index([role])
}

enum Role {
  ADMIN
  OPERATOR
}

model Guide {
  id                    String   @id @default(uuid())
  guideNumber           String   @unique
  razonSocial           String
  localidad             String
  direccion             String
  identificacionUsuario String
  referenciaEntrega     String?
  fechaEntrega          DateTime?
  horaEntrega           String?
  estado                String   @default("PENDIENTE")
  tipoInmueble          String?
  piso                  String?
  tipoPuerta            String?
  nombreRecibe          String?
  ccRecibe              String?
  nombreEntrega         String?
  observations          String?
  barcode               String?
  userId                String
  user                  User     @relation(fields: [userId], references: [id])
  createdAt             DateTime @default(now())
  updatedAt             DateTime @updatedAt
  
  @@index([guideNumber])
  @@index([userId])
  @@index([createdAt])
  @@index([estado])
}

model Setting {
  id            String   @id @default(uuid())
  key           String   @unique
  value         String?
  type          String   @default("string")
  description   String?
  updatedAt     DateTime @updatedAt
  
  @@index([key])
}

model GuideConfig {
  id              String   @id @default(uuid())
  prefix          String   @default("GUIA")
  startNumber     Int      @default(1)
  currentNumber   Int      @default(1)
  endNumber       Int?
  digitos         Int      @default(6)
  active          Boolean  @default(true)
  logo            String?
  primaryColor    String   @default("#1e40af")
  secondaryColor  String   @default("#3b82f6")
  companyName     String   @default("Mekaplus Express")
  companyNIT      String   @default("901505437-1")
  companyAddress  String?
  companyPhone    String?
  updatedAt       DateTime @updatedAt
}

model AuditLog {
  id        String   @id @default(uuid())
  action    String
  table     String
  recordId  String?
  userId    String?
  details   Json?
  createdAt DateTime @default(now())
  
  @@index([userId])
  @@index([createdAt])
  @@index([action])
}
